  <script>
    const form = document.getElementById('ventaForm');
    const cantidad = document.getElementById('cantidad');
    const precio = document.getElementById('precio');
    const precioIGV = document.getElementById('precioIGV');
    const totalIGV = document.getElementById('totalIGV');

    function calcular() {
      const cant = parseFloat(cantidad.value) || 0;
      const p = parseFloat(precio.value) || 0;
      const pIGV = p * 1.18;
      const tIGV = cant * pIGV;
      precioIGV.value = pIGV.toFixed(2);
      totalIGV.value = tIGV.toFixed(2);
    }

    cantidad.addEventListener('input', calcular);
    precio.addEventListener('input', calcular);

    form.addEventListener('submit', async (e) => {
      e.preventDefault();
      const data = Object.fromEntries(new FormData(form).entries());
      data.precioIGV = precioIGV.value;
      data.totalIGV = totalIGV.value;
      data.fecha = new Date().toLocaleDateString();

      // üîó Aqu√≠ colocas el link de tu flujo Power Automate:
      const url = "AQU√ç_VA_TU_LINK_WEBHOOK";

      await fetch(url, {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify(data),
      });

      alert("‚úÖ Venta registrada correctamente");
      form.reset();
      precioIGV.value = '';
      totalIGV.value = '';
    });
  </script>
